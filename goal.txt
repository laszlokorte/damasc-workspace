template example(minvalue is Integer, maxcount is Integer) {

	init {
		put W: init someNet(2,3)
	}

	bag W { }
	bag A { 1;2;3; }
	bag B { 1;2;3; }
	bag C { 1;2;3; }

	alias bag A as AAAA
	alias action bar as baaar

	@interface(in=bazing) 
	bag L { 42 }

	@interface(out=chaka) 
	bag N { true }

	bag R { 1;2;3 }

	bag S {1;2;3} only x where x < 10 limit maxcount
	bag T {1;2;3} only x where x < 10 limit maxcount


	action foo(a, b) {
		take A: a;
		put B: b;
	}

	@interface(in=baz) 
	action bar {
		take A: x;y where x != y;
		put B: x+y;
		put C: somevar;
		test L: 42
		test not N: true

		take all S: lst
		put many T: lst
		take many T: lst


		reserve R: p;q where p > q

		reserve W: otherNet;

		trigger otherNet.doStuff;
		trigger foo(2, 3);

		emit {event: "foo", payload: somevar} if x==20;

		guard x + y > 20;
		guard y >= minvalue;

		with somevar=23*x+y;

		return [x, y];
	}

	@manual
	action debug {
		put L: 42
	}
}
